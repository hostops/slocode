CREATE TABLE `post` (  `ID_Post` varchar(36) COLLATE utf8_unicode_ci NOT NULL,  `ID_Topic` varchar(36) COLLATE utf8_unicode_ci NOT NULL,  `ID_Author` varchar(36) COLLATE utf8_unicode_ci NOT NULL,  `Date` datetime NOT NULL,  `Content` mediumtext CHARACTER SET utf32 COLLATE utf32_unicode_ci NOT NULL,  `Content_text` mediumtext CHARACTER SET utf32 COLLATE utf32_unicode_ci NOT NULL,  PRIMARY KEY (`ID_Post`),  UNIQUE KEY `ID_Topic` (`ID_Topic`,`ID_Author`)) ENGINE=MyISAM DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci
