CREATE TABLE `thread` (  `ID_Thread` varchar(36) CHARACTER SET utf8 COLLATE utf8_slovenian_ci NOT NULL,  `ID_Author` varchar(36) CHARACTER SET utf8 COLLATE utf8_slovenian_ci NOT NULL,  `ID_Category` varchar(36) CHARACTER SET utf8 COLLATE utf8_slovenian_ci NOT NULL,  `Date` datetime NOT NULL,  `Title` varchar(2048) CHARACTER SET utf32 COLLATE utf32_unicode_ci NOT NULL,  `Content` mediumtext CHARACTER SET utf32 COLLATE utf32_unicode_ci NOT NULL,  `Content_text` mediumtext CHARACTER SET utf32 COLLATE utf32_unicode_ci NOT NULL,  `Tags` varchar(256) CHARACTER SET utf32 COLLATE utf32_unicode_ci NOT NULL,  PRIMARY KEY (`ID_Thread`)) ENGINE=MyISAM DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci
